JavaScript code

document.getElementById("sortBtn").addEventListener("click",sort);
function sort() {
    let table = document.getElementById("myTable");
    let switching = true;
    let shouldSwitch;
    while (switching) {
      switching = false;
      rows = table.rows;
      let i;
      for (i = 1; i < (rows.length - 1); i++) {
        shouldSwitch = false;
        let x = rows[i].getElementsByTagName("TD")[0];
        let y = rows[i + 1].getElementsByTagName("TD")[0];
        if (x.innerHTML.toLowerCase() > y.innerHTML.toLowerCase()) {
          shouldSwitch = true;
         break;
        }
      }
      if (shouldSwitch) {
        let rows[i].parentNode.insertBefore(rows[i + 1], rows[i]);
        switching = true;
      }
    }
}
