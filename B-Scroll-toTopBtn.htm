This is an example that shows a Button when the screen is active onscroll.
You can copy the code below.

Code:

<!DOCTYPE html>
<html>
    <head>
        <meta charset="utf-8">
        <meta name="viewport" content="device-width, initial-scale=1">
        <style>
          #Deon{
            padding:20px;
            background:blue;
            color:white;
            border-radius:10px;
            border:none;
            position:absolute;
            top:400px;
            left:250px;
            display:none;
          }
          #Deon:hover{
background:royalblue;
          }
        </style>
    </head>
    <body>
        <button id="Deon">Deon</button>
        <script>
          let Deon = document.getElementById ("Deon");
          window.onscroll = function() {scrollFunction()};
          function scrollFunction(){
            if (document.body.scrollTop > 20 ||
              document.documentElement.scrollTop > 20){
              Deon.style.display = "block";
            } else{
              Deon.style.display = "none";
            }
          }
        </script>
    </body>
</html>
