JavaScript code

document.getElementById("sortBtn").addEventListener('click',sort);
function sort(){
    let ul=document.getElementById("list");
    let switching=true;
    let shouldSwitch;
    while(switching){
      switching=false;
      let li=ul.getElementsByTagName("li");
      let i;
      for(i=0; i<(li.length-1);i++){
       shouldSwitch=false;
       if(li[i].innerHTML.toLowerCase() > li[i + 1].innerHTML.toLowerCase()){
          shouldSwitch =true;
          break;
       }
      }
      if(shouldSwitch){
       li[i].parentNode.insertBefore(li[i+1],li[i]);
        switching=true;
      }
    }
}
