JavaScript code

document.getElementById("myInput").addEventListener("keyup",myFilter);
function myFilter(){
    let input =document.getElementById("myInput");
    let x =input.value.toUpperCase();
    let myUL =document.getElementById("unL");
    let LI =myUL.getElementsByTagName("li");
    let i;
    for(i=0; i < LI.length; i++){
        let a=LI[i].getElementsByTagName("a")[0];
        let txtValue =a.innerText || a.textContent;
        if(txtValue.toUpperCase().indexOf(x) > -1){
          LI[i].style.display="";
        }else{
          LI[i].style.display="none";
        }
    }
}
