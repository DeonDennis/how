JavaScript code

function Move(){
    let id =null;
    let anim =document.getElementById("Animation");
    let pos =0;
    clearInterval(id);
    id=setInterval(moveFrame,5);
    function moveFrame(){
      if(pos == 425){
clearInterval(id);
      }else{
pos++;
        anim.style.top=pos + 'px';
        anim.style.left=pos + 'px';
      }
    }
}
