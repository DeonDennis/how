let output= document.getElementById("output");
function display(v){
    output.value += v;
}
function solve(){
    output.value=eval(output.value);
    if(output == ""){
      output.value="";
    }
}
function ClearOutput(){
    output.value=output.value.substr(0,output.value.length-1);
}
